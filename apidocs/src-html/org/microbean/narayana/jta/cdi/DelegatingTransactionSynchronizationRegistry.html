017package org.microbean.narayana.jta.cdi;
018
019import java.util.Map; // for javadoc only
020import java.util.Objects;
021
022import javax.transaction.Status; // for javadoc only
023import javax.transaction.Synchronization;
024import javax.transaction.Transaction; // for javadoc only
025import javax.transaction.TransactionManager; // for javadoc only
026import javax.transaction.TransactionSynchronizationRegistry;
027
028/**
029 * An {@code abstract} {@link TransactionSynchronizationRegistry}
030 * implementation that delegates all method invocations to another
031 * {@link TransactionSynchronizationRegistry}.
032 *
033 * <h2>Design Notes</h2>
034 *
035 * <p>This class is {@code public} for convenience.  It is extended by
036 * other non-{@code public} internal classes.</p>
037 *
038 * @author <a href="https://about.me/lairdnelson"
039 * target="_parent">Laird Nelson</a>
040 *
041 * @see TransactionSynchronizationRegistry
042 */
043public abstract class DelegatingTransactionSynchronizationRegistry implements TransactionSynchronizationRegistry {
044
045  private final TransactionSynchronizationRegistry delegate;
046
047  /**
048   * Creates a new {@link DelegatingTransactionSynchronizationRegistry}.
049   *
050   * @param delegate the {@link TransactionSynchronizationRegistry} to
051   * which all method invocations will be delegated; may be {@code
052   * null} in which case every method in this class will throw an
053   * {@link IllegalStateException} when invoked
054   *
055   */
056  protected DelegatingTransactionSynchronizationRegistry(final TransactionSynchronizationRegistry delegate) {
057    super();
058    this.delegate = delegate;
059  }
060
061  /**
062   * Return an opaque object to represent the transaction bound to the
063   * current thread at the time this method is called.
064   *
065   * <p>This method may return {@code null}.</p>
066   *
067   * <p>This object overrides {@link Object#hashCode()} and {@link
068   * Object#equals(Object)} to allow its use as the key in a {@link
069   * Map} for use by the caller. If there is no transaction currently
070   * active, this method will return {@code null}.</p>
071   *
072   * <p>The {@link Object} returned will return the same hashCode and
073   * compare equal to all other objects returned by calling this
074   * method from any component executing in the same transaction
075   * context in the same application server.</p>
076   *
077   * <p>The {@link Object#toString()} method returns a {@link String}
078   * that might be usable by a human reader to usefully understand the
079   * transaction context. The {@link Object#toString()} result is
080   * otherwise not defined. Specifically, there is no forward or
081   * backward compatibility guarantee of the results of the returned
082   * {@link Object}'s {@link Object#toString()} override.</p>
083   * 
084   * <p>The object is not necessarily serializable, and has no defined
085   * behavior outside the virtual machine whence it was obtained.</p>
086   *
087   * @return an opaque object representing the transaction bound to
088   * the current thread at the time this method is called, or {@code
089   * null}
090   *
091   * @exception IllegalStateException if a {@code null} {@code
092   * delegate} was supplied at {@linkplain
093   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
094   * construction time}
095   */
096  @Override
097  public Object getTransactionKey() {
098    if (this.delegate == null) {
099      throw new IllegalStateException("delegate == null");
100    }
101    return this.delegate.getTransactionKey();
102  }
103
104  /**
105   * Adds or replaces an object in the {@link Map} of resources being
106   * managed for the transaction bound to the current thread at the
107   * time this method is called.
108   *
109   * <p>The supplied key should be of an caller-defined class so as
110   * not to conflict with other users. The class of the key must
111   * guarantee that the {@link Object#hashCode() hashCode()} and
112   * {@link Object#equals(Object) equals(Object)} methods are suitable
113   * for use as keys in a {@link Map}. The key and value are not
114   * examined or used by the implementation. The general contract of
115   * this method is that of {@link Map#put(Object, Object)} for a
116   * {@link Map} that supports non-{@code null} keys and null
117   * values. For example, if there is already an value associated with
118   * the key, it is replaced by the {@code value} parameter.</p>
119   *
120   * @param key the key for the {@link Map} entry; must not be {@code null}
121   *
122   * @param value the value for the {@link Map} entry
123   *
124   * @exception IllegalStateException if no transaction is active or
125   * if a {@code null} {@code delegate} was supplied at {@linkplain
126   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
127   * construction time}
128   *
129   * @exception NullPointerException if the parameter {@code key} is
130   * {@code null}
131  */
132  @Override
133  public void putResource(final Object key, final Object value) {
134    if (this.delegate == null) {
135      throw new IllegalStateException("delegate == null");
136    }
137    this.delegate.putResource(key, value);
138  }
139
140  /**
141   * Gest an object from the {@link Map} of resources being managed
142   * for the transaction bound to the current thread at the time this
143   * method is called.
144   *
145   * <p>The key should have been supplied earlier by a call to {@link
146   * #putResource(Object, Object)} in the same transaction. If the key
147   * cannot be found in the current resource {@link Map}, {@code null}
148   * is returned. The general contract of this method is that of
149   * {@link Map#get(Object)} for a {@link Map} that supports
150   * non-{@code null} keys and null values. For example, the returned
151   * value is null if there is no entry for the parameter {@code key}
152   * or if the value associated with the key is actually {@code
153   * null}.</p>
154   *
155   * @param key the key for the {@link Map} entry
156   *
157   * @return the value associated with the supplied {@code key}; may
158   * be {@code null}
159   *
160   * @exception IllegalStateException if no transaction is active or
161   * if a {@code null} {@code delegate} was supplied at {@linkplain
162   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
163   * construction time}
164   *
165   * @exception NullPointerException if the parameter {@code key} is
166   * {@code null}
167   */
168  @Override
169  public Object getResource(final Object key) {
170    if (this.delegate == null) {
171      throw new IllegalStateException("delegate == null");
172    }
173    return this.delegate.getResource(key);
174  }
175
176  /**
177   * Registers a {@link Synchronization} instance with special ordering semantics.
178   *
179   * <p>The supplied {@link Synchronization}'s {@link
180   * Synchronization#beforeCompletion()} method will be called after
181   * all {@code SessionSynchronization#beforeCompletion()} callbacks
182   * and callbacks registered directly with the {@link Transaction},
183   * but before the 2-phase commit process starts. Similarly, the
184   * {@link Synchronization#afterCompletion(int)} callback will be
185   * called after 2-phase commit completes but before any {@code
186   * SessionSynchronization} and {@link Transaction} {@code
187   * afterCompletion(int)} callbacks.</p>
188   *
189   * <p>The {@link Synchronization#beforeCompletion()} callback will
190   * be invoked in the transaction context of the transaction bound to
191   * the current thread at the time this method is called. Allowable
192   * methods include access to resources, e.g. connectors. No access
193   * is allowed to "user components" (e.g. timer services or bean
194   * methods), as these might change the state of data being managed
195   * by the caller, and might change the state of data that has
196   * already been flushed by another caller of {@link
197   * #registerInterposedSynchronization(Synchronization)}. The general
198   * context is the component context of the caller of {@link
199   * #registerInterposedSynchronization(Synchronization)}.</p>
200   *
201   * <p>The {@link Synchronization#afterCompletion(int)} callback will
202   * be invoked in an undefined context. No access is permitted to
203   * "user components" as defined above. Resources can be closed but
204   * no transactional work can be performed with them.</p>
205   *
206   * <p>If this method is invoked without an active transaction
207   * context, an {@link IllegalStateException} is thrown.</p>
208   *
209   * <p>If this method is invoked after the two-phase commit
210   * processing has started, an {@link IllegalStateException} is
211   * thrown.</p>
212   *
213   * @param synchronization the {@link Synchronization} to register;
214   * must not be {@code null}
215   *
216   * @exception IllegalStateException if no transaction is active or
217   * two-phase commit processing has started or if a {@code null}
218   * {@code delegate} was supplied at {@linkplain
219   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
220   * construction time}
221   *
222   * @see Synchronization
223   *
224   * @see Synchronization#beforeCompletion()
225   *
226   * @see Synchronization#afterCompletion(int)
227   */
228  @Override
229  public void registerInterposedSynchronization(final Synchronization synchronization) {
230    if (this.delegate == null) {
231      throw new IllegalStateException("delegate == null");
232    }
233    this.delegate.registerInterposedSynchronization(synchronization);
234  }
235
236  /**
237   * Return the status of the transaction bound to the current thread
238   * at the time this method is called.
239   *
240   * <p>This is the result of executing {@link
241   * TransactionManager#getStatus()} in the context of the transaction
242   * bound to the current thread at the time this method is
243   * called.</p>
244   *
245   * @return the status of the transaction bound to the current thread
246   * at the time this method is called; will be equal the value of one
247   * of the constants defined in the {@link Status} class
248   *
249   * @exception IllegalStateException if a {@code null} {@code
250   * delegate} was supplied at {@linkplain
251   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
252   * construction time}
253   *
254   * @see TransactionManager#getStatus()
255   *
256   * @see Status
257   */
258  @Override
259  public int getTransactionStatus() {
260    if (this.delegate == null) {
261      throw new IllegalStateException("delegate == null");
262    }
263    return this.delegate.getTransactionStatus();
264  }
265
266  /**
267   * Sets the {@code rollbackOnly} status of the transaction bound to
268   * the current thread at the time this method is called.
269   *
270   * @exception IllegalStateException if no transaction is active or
271   * if a {@code null} {@code delegate} was supplied at {@linkplain
272   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
273   * construction time}
274   */
275  @Override
276  public void setRollbackOnly() {
277    if (this.delegate == null) {
278      throw new IllegalStateException("delegate == null");
279    }
280    this.delegate.setRollbackOnly();
281  }
282
283  /**
284   * Get the {@code rollbackOnly} status of the transaction bound to
285   * the current thread at the time this method is called.
286   *
287   * @return the {@code rollbackOnly} status
288   *
289   * @exception IllegalStateException if no transaction is active or
290   * if a {@code null} {@code delegate} was supplied at {@linkplain
291   * #DelegatingTransactionSynchronizationRegistry(TransactionSynchronizationRegistry)
292   * construction time}
293   */
294  @Override
295  public boolean getRollbackOnly() {
296    if (this.delegate == null) {
297      throw new IllegalStateException("delegate == null");
298    }
299    return this.delegate.getRollbackOnly();
300  }
